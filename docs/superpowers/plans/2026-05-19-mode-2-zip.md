# Mode-2 ZIP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `profile` mode of the `zip/` framework into a recovery-flashable, self-contained per-OTA mode-2 installer: detect OEM from `build.prop`, derive the mode-2 profile on-device from the stock vbmeta, OEM-patch the cached ABL, pack the `cached_abl + mode2_profile` overlay onto `mode-2.efi`, write EFISP + loader ABL.

**Architecture:** Three slices. (1) The profile intermediate moves to a flat TOML format; a new multi-platform C `mode2-profile` tool (`derive`+`compile`, vendored `tomlc99`) joins the Python tool. (2) `abl-patcher` gains runtime patch-scope selection (`--oem`). (3) `install.sh`'s loader-ABL machinery is factored into `zip/core/` and the `profile` mode is built on it.

**Tech Stack:** C99 (recovery tools, cross-compiled aarch64 via NDK), Python 3 (`tomllib`), busybox `ash` (ZIP scripts), the EDK2 Docker build.

**Spec:** `docs/superpowers/specs/2026-05-19-mode-2-zip-design.md`. Work branches from `main`. The `zip/` directory is a git submodule (`zip-gbl-chainload`) ŌĆö changes there are commits *in the submodule* plus a pointer bump in the parent.

```python
M2P_KEYS = {"version", "is_unlocked", "color", "system_version",
            "system_spl", "rot_digest", "pubkey_digest", "vbh"}


def _req(d: dict, key: str, typ):
    if key not in d:
        raise SystemExit(f"error: '{key}' missing from profile")
    v = d[key]
    if not isinstance(v, typ):
        raise SystemExit(f"error: '{key}' has wrong type (expected {typ.__name__})")
    return v


def _digest(d: dict, key: str) -> bytes:
    s = _req(d, key, str)
    if len(s) != 64 or re.fullmatch(r"[0-9a-f]{64}", s) is None:
        raise SystemExit(f"error: '{key}' must be 64 lowercase-hex chars")
    return bytes.fromhex(s)


def cmd_compile(args) -> int:
    toml_path = Path(args.toml)
    if not toml_path.is_file():
        raise SystemExit(f"error: profile not found: {toml_path}")
    try:
        prof = tomllib.loads(toml_path.read_text())
    except tomllib.TOMLDecodeError as e:
        raise SystemExit(f"error: malformed profile TOML: {e}")

    unknown = set(prof) - M2P_KEYS
    if unknown:
        raise SystemExit(f"error: unknown key(s) in profile: {', '.join(sorted(unknown))}")

    if _req(prof, "version", int) != 1:
        raise SystemExit(f"error: profile version {prof['version']}, expected 1")
    is_unlocked    = _req(prof, "is_unlocked", int)
    color          = _req(prof, "color", int)
    system_version = _req(prof, "system_version", int)
    system_spl     = _req(prof, "system_spl", int)
    rot_digest     = _digest(prof, "rot_digest")
    pubkey_digest  = _digest(prof, "pubkey_digest")
    vbh            = _digest(prof, "vbh")

    if is_unlocked not in (0, 1):
        raise SystemExit(f"error: is_unlocked must be 0 or 1 (got {is_unlocked})")
    if color not in (0, 1, 2, 3):
        raise SystemExit(f"error: color must be 0..3 (got {color})")
    for name, v in (("system_version", system_version), ("system_spl", system_spl)):
        if not (0 <= v <= 0xFFFFFFFF):
            raise SystemExit(f"error: {name} out of u32 range (got {v})")

    blob = struct.pack(M2P_STRUCT, M2P_MAGIC, M2P_VERSION, 0,
                       is_unlocked, color, system_version, system_spl,
                       rot_digest, pubkey_digest, vbh)
    assert len(blob) == M2P_SIZE
    Path(args.output).write_bytes(blob)
    print(f"wrote {args.output} ({len(blob)} bytes)")
    return 0
```

```c
/* tools/mode2-profile/mode2-profile.c ŌĆö C mode-2 profile tool.
   derive: vbmeta.img -> profile.toml   (Task 3)
   compile: profile.toml -> 120-byte gbl_mode2_profile binary. */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>
#include "vendor/tomlc99/toml.h"
#include "../shared/gbl_mode2_profile.h"

static void wle16(uint8_t *p, uint16_t v){p[0]=v;p[1]=v>>8;}
static void wle32(uint8_t *p, uint32_t v){p[0]=v;p[1]=v>>8;p[2]=v>>16;p[3]=v>>24;}

/* fail: print "error: ..." to stderr and exit 1. No partial output. */
static void fail(const char *msg){ fprintf(stderr,"error: %s\n",msg); exit(1); }

/* hexkey: read a TOML string key, require exactly 64 lowercase-hex, decode
   into out[32]. */
static void hexkey(toml_table_t *t, const char *key, uint8_t out[32]) {
    toml_datum_t d = toml_string_in(t, key);
    if (!d.ok) { fprintf(stderr,"error: '%s' missing or not a string\n",key); exit(1); }
    if (strlen(d.u.s) != 64) { free(d.u.s);
        fprintf(stderr,"error: '%s' must be 64 hex chars\n",key); exit(1); }
    for (int i=0;i<32;i++){
        int hi=d.u.s[2*i], lo=d.u.s[2*i+1];
        if(!( (hi>='0'&&hi<='9')||(hi>='a'&&hi<='f') ) ||
           !( (lo>='0'&&lo<='9')||(lo>='a'&&lo<='f') )) { free(d.u.s);
            fprintf(stderr,"error: '%s' contains non-lowercase-hex\n",key); exit(1); }
        out[i] = (uint8_t)((( hi<='9'?hi-'0':hi-'a'+10)<<4)|(lo<='9'?lo-'0':lo-'a'+10));
    }
    free(d.u.s);
}

/* intkey: read a TOML integer key, enforce [lo,hi]. */
static int64_t intkey(toml_table_t *t, const char *key, int64_t lo, int64_t hi) {
    toml_datum_t d = toml_int_in(t, key);
    if (!d.ok) { fprintf(stderr,"error: '%s' missing or not an integer\n",key); exit(1); }
    if (d.u.i < lo || d.u.i > hi) {
        fprintf(stderr,"error: '%s' out of range %lld..%lld (got %lld)\n",
                key,(long long)lo,(long long)hi,(long long)d.u.i); exit(1); }
    return d.u.i;
}

static int do_compile(const char *in, const char *out) {
    FILE *f = fopen(in,"r");
    if (!f) { perror(in); return 1; }
    char errbuf[200];
    toml_table_t *t = toml_parse_file(f, errbuf, sizeof errbuf);
    fclose(f);
    if (!t) { fprintf(stderr,"error: malformed profile TOML: %s\n",errbuf); return 1; }

    /* reject unknown keys */
    static const char *known[] = {"version","is_unlocked","color","system_version",
        "system_spl","rot_digest","pubkey_digest","vbh"};
    for (int i=0;; i++) {
        const char *k = toml_key_in(t, i);
        if (!k) break;
        int ok=0; for (unsigned j=0;j<sizeof known/sizeof*known;j++)
            if(!strcmp(k,known[j])) ok=1;
        if(!ok){ fprintf(stderr,"error: unknown key '%s' in profile\n",k);
                 toml_free(t); return 1; }
    }

    if (intkey(t,"version",1,1) != 1) fail("version must be 1");
    uint32_t is_unlocked    = (uint32_t)intkey(t,"is_unlocked",0,1);
    uint32_t color          = (uint32_t)intkey(t,"color",0,3);
    uint32_t system_version = (uint32_t)intkey(t,"system_version",0,0xFFFFFFFFLL);
    uint32_t system_spl     = (uint32_t)intkey(t,"system_spl",0,0xFFFFFFFFLL);
    uint8_t rot[32], pk[32], vbh[32];
    hexkey(t,"rot_digest",rot);
    hexkey(t,"pubkey_digest",pk);
    hexkey(t,"vbh",vbh);
    toml_free(t);

    uint8_t b[GBL_M2P_SIZE];
    memset(b,0,sizeof b);
    memcpy(b+0, GBL_M2P_MAGIC, 4);
    wle16(b+4, GBL_M2P_VERSION);
    /* b+6 reserved = 0 */
    wle32(b+8,  is_unlocked);
    wle32(b+12, color);
    wle32(b+16, system_version);
    wle32(b+20, system_spl);
    memcpy(b+24, rot, 32);
    memcpy(b+56, pk,  32);
    memcpy(b+88, vbh, 32);

    FILE *o = fopen(out,"wb");
    if (!o) { perror(out); return 1; }
    if (fwrite(b,1,sizeof b,o)!=sizeof b){ fclose(o);
        fprintf(stderr,"error: write failed\n"); return 1; }
    fclose(o);
    fprintf(stderr,"wrote %s (%u bytes)\n", out, (unsigned)sizeof b);
    return 0;
}

int derive_main(int argc, char **argv);  /* Task 3, in this file */

int main(int argc, char **argv) {
    if (argc >= 5 && !strcmp(argv[1],"compile") && !strcmp(argv[3],"-o"))
        return do_compile(argv[2], argv[4]);
    if (argc >= 2 && !strcmp(argv[1],"derive"))
        return derive_main(argc, argv);
    fprintf(stderr,
      "usage: mode2-profile compile <in.toml> -o <out.bin>\n"
      "       mode2-profile derive  <vbmeta.img> -o <out.toml>\n");
    return 2;
}
```

- [ ] **Step 1: Implement `derive_main` ŌĆö the AVB vbmeta reader**

Add `derive_main` to `mode2-profile.c`. The AVB vbmeta format (cross-reference: the Python `_read_pubkey_blob` / `_parse_props` / `_compute_vbh` in `mode2-profile.py`, and `avbtool.py`'s `AvbVBMetaHeader`):

- The image is `header(256) ŌĆ¢ authentication_block ŌĆ¢ auxiliary_block`. The header is **big-endian**. Fields needed, by byte offset into the 256-byte header:
  - `+0` magic `"AVB0"` (4 bytes) ŌĆö verify.
  - `+12` `authentication_data_block_size` (u64 BE).
  - `+20` `auxiliary_data_block_size` (u64 BE).
  - `+64` `public_key_offset` (u64 BE) ŌĆö offset *within* the auxiliary block.
  - `+72` `public_key_size` (u64 BE).
  - `+96` `descriptors_offset`, `+104` `descriptors_size` (u64 BE).
  (Offsets per `avbtool.py`'s `AvbVBMetaHeader` FORMAT_STRING ŌĆö `4s` magic, `2L`
  required-libavb-version, then the `Q` fields. Verify against `avbtool.py`.)
- `aux_off = 256 + authentication_data_block_size`. The public key blob is `image[aux_off + public_key_offset .. + public_key_size]`. Bounds-check every offset against the file size; `public_key_size == 0` ŌåÆ `fail("vbmeta has no public key (unsigned?)")`.
- Compute: `rot_digest = SHA256(pubkey ŌĆ¢ 0x00)`, `pubkey_digest = SHA256(pubkey)`, `vbh = SHA256(image[0 .. 256 + auth_size + aux_size])`.
- **Descriptors** live in the auxiliary block. Walk them: each descriptor begins with `tag` (u64 BE) and `num_bytes_following` (u64 BE); step `16 + round_up(num_bytes_following, 8)`... ŌĆö read `avbtool.py`'s `parse_descriptors` for the exact stride. Property descriptors have `tag == 0`; their body is `key_num_bytes`(u64 BE), `value_num_bytes`(u64 BE), then `key ŌĆ¢ 0x00 ŌĆ¢ value ŌĆ¢ 0x00`. Find the properties `com.android.build.boot.os_version` and `com.android.build.boot.security_patch`; absent ŌåÆ `fail`.
- Encode (identical formulas to the Python `_encode_os_version`/`_encode_spl`): `os_version = (major<<14)|(minor<<7)|sub`; `spl = (day<<11)|((year-2000)<<4)|month`.
- Reuse a SHA-256 implementation: `GblChainloadPkg/Library/GblPayloadLib/Sha256.c` exposes `gbl_sha256` and builds host-side with `GBL_HOST_BUILD` ŌĆö compile it into the tool (add to the Makefile `SRCS`) rather than hand-rolling SHA-256.

Write the `.toml` with the **exact same byte layout** as the Python `cmd_derive` (Task 1 Step 2) ŌĆö same comment lines, same key order, same `0x%x` / quoted-hex formatting ŌĆö so the parity test's `cmp` of the two `.toml` files passes.

```c
   The on-device EDK2 build uses DynamicPatchLib_EnsureInit() (GBL_MODE

#ifndef GBL_PATCH_SCOPE_H_
#define GBL_PATCH_SCOPE_H_
```

- [ ] **Step 2: Add the runtime aggregator to `PatchTable.c`**

```c
#include "PatchScope.h"

void
DynamicPatchLib_EnsureInitScoped (GBL_OEM oem, int include_mode1)
{
  UINTN n = 0, i;
  for (i = 0; i < kUniversalPatchesCount && n < MAX_PATCHES; ++i)
    gAggregated[n++] = kUniversalPatches[i];
  if (oem == GBL_OEM_ONEPLUS)

  gAggregatedLen = n;
  gPatchTable = gAggregated;

}
```

```sh
# shellcheck shell=sh
# shellcheck disable=SC2154,SC2015
# modes/profile.sh ŌĆö install gbl-chainload mode-2.

STOCK_VBMETA=/sdcard/stock_vbmeta.img
PROFILE_TOML=/sdcard/gbl-chainload_profile.toml

# detect_oem -> sets OEM_ID. Reads build.prop from a mounted system/vendor.
detect_oem() {
  _bp=""
  for _p in /system/system/build.prop /system/build.prop /vendor/build.prop; do
    [ -f "$_p" ] && { _bp="$_p"; break; }
  done
  [ -n "$_bp" ] || abort "build.prop not found (mount system/vendor first)"
  _mfr=$(grep -m1 '^ro.product.*manufacturer=' "$_bp" | cut -d= -f2 | tr 'A-Z' 'a-z')
  case "$_mfr" in
    *oneplus*|*oppo*) OEM_ID=oneplus ;;
    *) abort "unsupported OEM (build.prop manufacturer='$_mfr')" ;;
  esac
  ui_print "[*] OEM detected: $OEM_ID"
}

profile_preflight() {
  [ -n "$SLOT" ] && [ -n "$INACTIVE" ] || abort "not an A/B device"
  [ -f "$STOCK_VBMETA" ] || abort "$STOCK_VBMETA required ŌĆö place the stock OEM vbmeta there"
  EFISP_DEV=$(byname efisp); [ -n "$EFISP_DEV" ] || abort "efisp not found"
  _mz=$(dd if="$EFISP_DEV" bs=1 count=2 2>/dev/null | od -An -tx1 | tr -d ' \n')
  [ "$_mz" = 4d5a ] || abort "EFISP does not hold a PE (got '$_mz')"
}

build_profile() {
  ui_print "[1/5] deriving mode-2 profile from stock vbmeta"
  mode2-profile derive "$STOCK_VBMETA" -o "$PROFILE_TOML" \
    || abort "mode2-profile derive failed"
  ui_print "[2/5] compiling profile"
  mode2-profile compile "$PROFILE_TOML" -o "$WORKDIR/profile.bin" \
    || abort "mode2-profile compile failed"
}

build_m2_payload() {
  ui_print "[3/5] caching abl_$TARGET (OEM-patched, --oem $OEM_ID)"
  dd if="$TARGET_DEV" of="$WORKDIR/cache_abl.img" bs=1M 2>/dev/null \
    || abort "failed to read abl_$TARGET"
  fv-unwrap "$WORKDIR/cache_abl.img" "$WORKDIR/extracted.efi" >/dev/null 2>&1 \
    || abort "fv-unwrap failed"
  abl-patcher --in "$WORKDIR/extracted.efi" --out "$WORKDIR/patched.efi" \
    --oem "$OEM_ID" --no-mode1 || abort "abl-patcher failed"
  ui_print "[4/5] packing GBLP1 overlay (cached_abl + mode2_profile)"
  gbl-pack --cached-abl "$WORKDIR/patched.efi" \
           --source "$WORKDIR/cache_abl.img" --extracted "$WORKDIR/extracted.efi" \
           --mode2-profile "$WORKDIR/profile.bin" \
           --out "$WORKDIR/payload.bin" || abort "gbl-pack failed"
  cat "$WORKDIR/base/mode-2.efi" "$WORKDIR/payload.bin" > "$WORKDIR/installed.efi"
}

mode_main() {
  ui_print "profile: gbl-chainload mode-2 installer"
  ui_print ""
  detect_oem
  pick_scenario          # shared core: sets SCENARIO, TARGET
  profile_preflight
  TARGET_DEV=$(byname "abl_$TARGET"); [ -n "$TARGET_DEV" ] || abort "abl_$TARGET not found"
  [ -f "$WORKDIR/base/mode-2.efi" ] || abort "base/mode-2.efi missing from ZIP"
  resolve_restore_source # shared core: sets RESTORE_SRC
  build_profile
  build_m2_payload
  ui_print "[5/5] writing EFISP + loader ABL (backup + verify)"
  commit_verified "$WORKDIR/installed.efi" "$EFISP_DEV" /sdcard/efisp.bak
  restore_abl            # shared core
  save_backup_abl        # shared core
  ui_print ""
  ui_print "profile: done ŌĆö reboot. mode-2 active."
}
```

(`pick_scenario`/`resolve_restore_source`/`restore_abl`/`save_backup_abl`/`commit_verified` come from the shared `core/` libs. `byname` is from `core/partition.sh`. Confirm names against the actual `core/` files and adjust if they differ.)
